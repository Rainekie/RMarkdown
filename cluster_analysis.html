%% make tables
dat = readmatrix('Rstan_SDT/res_prod_SDT.csv');
dat = dat(:,2:end);
name = array2table([-150, -75, -60, -45, -30, 30, 45, 60, 75, 150]');
name.Properties.VariableNames = {'name'};
dat_80dB_d = array2table(dat(1:10,1:2:16));
dat_80dB_d = [name,dat_80dB_d];
dat_80dB_d.Properties.VariableNames = {'name' 's1' 's2' 's3' 's4' 's5' 's6' 's7' 's8'};
dat_80dB_sd = array2table(dat(1:10,2:2:16));
dat_80dB_sd = [name,dat_80dB_sd];
dat_80dB_sd.Properties.VariableNames = {'name' 's1' 's2' 's3' 's4' 's5' 's6' 's7' 's8'};
dat_68dB_d = array2table(dat(11:20,1:2:16));
dat_68dB_d = [name,dat_68dB_d];
dat_68dB_d.Properties.VariableNames = {'name' 's1' 's2' 's3' 's4' 's5' 's6' 's7' 's8'};
dat_68dB_sd = array2table(dat(11:20,2:2:16));
dat_68dB_sd = [name,dat_68dB_sd];
dat_68dB_sd.Properties.VariableNames = {'name' 's1' 's2' 's3' 's4' 's5' 's6' 's7' 's8'};
%% make plots
ti = {'80dB' '68dB'};
linescale = 0:0.01:6;
xre = -150:1:150;
dats1 = zeros(301,10);
dats2 = zeros(301,10);
for rep = 1:2
if rep == 1
plotdat = dat_80dB_d;
else
plotdat = dat_68dB_d;
end
x = table2array(plotdat(:,1));
y = table2array(plotdat(:,2));
underZero = find(y<=0);
y(underZero) = 0;
[obj, gof] = fit(x,y,'smoothingspline');
yhat = obj(xre);
if rep == 1
dats1(:,1:2) = [xre' yhat];
else
dats2(:,1:2) = [xre' yhat];
end
for i=1:7
x = table2array(plotdat(:,1));
y = table2array(plotdat(:,i+2));
underZero = find(y<=0);
y(underZero) = 0;
obj = fit(x,y,'smoothingspline');
yhat = obj(xre);
if rep == 1
dats1(:,i+2) = yhat;
else
dats2(:,i+2) = yhat;
end
end
end
%%
dats1(:,10) = mean(dats1(:,2:end),2);
dats2(:,10) = mean(dats2(:,2:end),2);
restable = zeros(8,3);
for i = 1:8
width1 = find(dats1(:,i+1) < 1);
width2 = find(dats2(:,i+1) < 1);
width1 = abs(dats1(width1(1),1))+dats1(width1(end),1);
width2 = abs(dats2(width2(1),1))+dats2(width2(end),1);
restable(i,1) = i;
restable(i,2) = width1;
restable(i,3) = width2;
end
for i = 1:8

hoge = abs(dats1(width1(1),1))+abs(dats1(width1(end),1));

end
